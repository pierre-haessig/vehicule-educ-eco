Difficulté : ces équa-diff sont données avec des conditions aux limites à la fois initiales et terminales.

# Description de l'équa-diff
def f(t,y):
    '''évolution dy/dt = f(t,y)'''
    # Séparation des composantes du vecteur d'état
    v,x,p1,p2 = y
    # pente
    s = 0
    dsdl = 0
    # force motrice optimale
    f_mot = -(m*v + p1)/(2*a*m)
    # Calcul des dérivés
    dvdt = (-c*v**2 - fr*np.sign(v) + f_mot - g*m*s)/m
    dxdt = v
    dp1dt = 2*c*p1*v/m - f_mot - p2
    dp2dt = g*p1*dsdl
    
    dydt = [dvdt, dxdt, dp1dt, dp2dt]
    return dydt

# Test de la dérivée à l'état initial:
print('pour y0 = %s, on a:' % y0)
f0 = f(t0, y0)
for i,yi in enumerate(['v', 'x', 'p_1', 'p_2']):
    display_latex(r'$$\frac{d%s}{dt}(0) = %f$$' % (yi, f0[i]), raw=True)

# Pas de discrétisation
N = 10
#T = T_max
T = 100 # [s]
# Vecteur temps:
t = np.linspace(t0, T, N+1)
dt = t[1] - t[0]
print('N=%d pts -> pas de temps %.1f s' % (N, dt))
print('t=%s (s)' % t)

# Solution de l'ODE:
y = np.zeros((N+1,4))*np.nan
y[0] = y0

i=0
r.set_initial_value(y0, t0)
print('Intégration...')
while r.t < T and i<=N-1:
    #print(i, r.t)
    i = i+1
    r.integrate(r.t+dt)
    if not r.successful():
        break
    y[i] = r.y
    imax = i

print(u'Succès de l\'intégration: %s' % r.successful())
print('dernier pas réussi: %d/%d' % (imax, N))
